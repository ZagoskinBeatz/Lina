# Текстовые редакторы — nano, vim, nvim

## Nano (простой)
```bash
nano file.txt

# Горячие клавиши
Ctrl+O          # сохранить
Ctrl+X          # выход
Ctrl+K          # вырезать строку
Ctrl+U          # вставить
Ctrl+W          # поиск
Ctrl+\          # поиск и замена
Ctrl+G          # справка
Alt+U           # отмена
Alt+E           # повтор
Ctrl+_          # перейти к строке/столбцу
Alt+A           # начать выделение
Alt+6           # копировать выделенное
Ctrl+T          # проверка орфографии
```

### Настройка nano
```bash
# ~/.nanorc или /etc/nanorc
set autoindent                 # авто-отступ
set tabsize 4                  # размер табуляции
set linenumbers                # номера строк
set mouse                      # поддержка мыши
set softwrap                   # перенос длинных строк
set constantshow               # показывать позицию курсора
include "/usr/share/nano/*.nanorc"  # подсветка синтаксиса
```

## Vim / Neovim
```bash
vim file.txt
nvim file.txt   # Neovim

# Режимы
i           # режим вставки (INSERT) — перед курсором
a           # режим вставки — после курсора
o           # новая строка ниже
O           # новая строка выше
R           # режим замены
Esc         # нормальный режим
v           # визуальный режим (выделение)
V           # визуальный (целые строки)
Ctrl+v      # блочное выделение (колонки)
:           # командный режим

# Выход
:q          # выход
:q!         # выход без сохранения
:wq         # сохранить и выйти
:x          # сохранить и выйти (если были изменения)
ZZ          # сохранить и выйти (быстро)
:wqa        # сохранить все буферы и выйти

# Навигация
h/j/k/l     # лево/вниз/вверх/право
w/b         # следующее/предыдущее слово
e           # конец слова
0/$         # начало/конец строки
^           # первый непробельный символ
gg/G        # начало/конец файла
42G         # перейти на строку 42
Ctrl+d/u    # полстраницы вниз/вверх
Ctrl+f/b    # полная страница вниз/вверх
%           # перейти к парной скобке
{/}         # предыдущий/следующий абзац
f<char>     # перейти к символу в строке
*           # поиск слова под курсором

# Редактирование
dd          # удалить строку
5dd         # удалить 5 строк
yy          # копировать строку
5yy         # копировать 5 строк
p           # вставить после
P           # вставить перед
u           # отмена
Ctrl+r      # повтор
.           # повторить последнее действие
ciw         # заменить слово (change inner word)
ci"         # заменить содержимое ""
ci(         # заменить содержимое ()
cit         # заменить содержимое тега
diw         # удалить слово
da"         # удалить включая ""
>>          # отступ вправо
<<          # отступ влево
J           # соединить строки
~           # сменить регистр
gU          # в верхний регистр
gu          # в нижний регистр
xp          # поменять 2 символа местами

# Поиск
/pattern    # поиск вперёд
?pattern    # поиск назад
n/N         # следующий/предыдущий результат
:%s/old/new/g  # замена во всём файле
:%s/old/new/gc # с подтверждением
:s/old/new/g   # замена в текущей строке

# Множественные файлы
:e file2    # открыть другой файл
:bn/:bp     # следующий/предыдущий буфер
:ls         # список буферов
:split      # горизонтальное разделение
:vsplit     # вертикальное разделение
Ctrl+w w    # переключение между окнами
Ctrl+w h/j/k/l  # навигация между окнами
:tabnew     # новая вкладка
gt/gT       # следующая/предыдущая вкладка

# Макросы
qa          # начать запись макроса в регистр a
q           # остановить запись
@a          # воспроизвести макрос a
10@a        # воспроизвести 10 раз

# Регистры (буферы обмена)
"ayy        # скопировать строку в регистр a
"ap         # вставить из регистра a
"+y         # скопировать в системный буфер
"+p         # вставить из системного буфера
:registers  # показать все регистры
```

## Neovim конфигурация
```lua
-- ~/.config/nvim/init.lua (базовая настройка)
vim.opt.number = true          -- номера строк
vim.opt.relativenumber = true  -- относительные номера
vim.opt.tabstop = 4            -- размер таба
vim.opt.shiftwidth = 4         -- размер отступа
vim.opt.expandtab = true       -- табы → пробелы
vim.opt.clipboard = "unnamedplus"  -- системный буфер обмена
vim.opt.ignorecase = true      -- регистронезависимый поиск
vim.opt.smartcase = true       -- умный RegCase при поиске
vim.opt.termguicolors = true   -- true color
vim.opt.scrolloff = 8          -- отступ от края при прокрутке
vim.opt.signcolumn = "yes"     -- столбец знаков всегда
vim.opt.undofile = true        -- persistent undo
vim.opt.updatetime = 250       -- быстрый CursorHold

-- Маппинги
vim.g.mapleader = " "          -- leader = пробел
vim.keymap.set('n', '<leader>w', ':w<CR>')     -- Space+w = сохранить
vim.keymap.set('n', '<leader>q', ':q<CR>')     -- Space+q = выйти
```

## Micro — современный терминальный редактор
```bash
sudo pacman -S micro

micro file.txt

# Управление как в обычном GUI-редакторе
Ctrl+S          # сохранить
Ctrl+Q          # выйти
Ctrl+C/V/X      # копировать/вставить/вырезать
Ctrl+Z          # отмена
Ctrl+F          # поиск
Ctrl+H          # замена
Ctrl+G          # помощь
Ctrl+E          # командная строка
```

## Helix — постмодерный редактор (Rust)
```bash
sudo pacman -S helix
hx file.txt

# Использует Selection → Action (выделить → действовать)
# w          # выделить слово
# d          # удалить выделенное
# c          # изменить выделенное
# y          # копировать
# p          # вставить
# :w         # сохранить
# :q         # выйти
# space+f    # поиск файлов (picker)
```
